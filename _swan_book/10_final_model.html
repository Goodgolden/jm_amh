10.1 Goal of the analysis

  • To separate the site and ethnicity as dummy variables

  • Fit the final joint model (without rate) with JMbayes2 based on model selection results

  • Dynamic prediction for the joint model with plots

  • Calculate the AUC and ROC for the joint model for \(Dt = 1, 2, 3, 4, 5\) (year)

10.4 Dynamic prediction

10.4.1 Individual prediction plots

Code
ids <- unique(swan_amh03$id)
dpplot <- function(id_num,
                   tmin = 0,
                   tmax = 12,
                   jmfit = jmbayes2.116,
                   ...){

  # browser()
  # ids <- unique(swan_amh03$id)
  nid0 <- swan_amh03[swan_amh03$id == ids[id_num], ]
  
  nid <- nid0
  nid <- nid[nid$time < tmin, ]
  nid$event <- 0
  nid$etime <- tmin
  
  Lpred <- predict(jmfit, 
                    newdata = nid,
                   times = seq(tmin, tmax, length.out = 11),
                    return_newdata = TRUE)
  Spred <- predict(jmfit, 
                 newdata = nid,
                 times = seq(tmin, tmax, length.out = 11),
                 process = "event",
                 return_newdata = TRUE)
  
  plot(Lpred, Spred, 
       ylab_long = paste0("ID_", id_num, "  log(amh)"))
  
  return(list(Lpred = Lpred,
              Spred = Spred,
              nid0 = nid0,
              nid = nid))
}

```{r}
#| label: "ddplot_function"
ids <- unique(swan_amh03$id)
dpplot <- function(id_num,
                   tmin = 0,
                   tmax = 12,
                   jmfit = jmbayes2.116,
                   ...){

  # browser()
  # ids <- unique(swan_amh03$id)
  nid0 <- swan_amh03[swan_amh03$id == ids[id_num], ]
  
  nid <- nid0
  nid <- nid[nid$time < tmin, ]
  nid$event <- 0
  nid$etime <- tmin
  
  Lpred <- predict(jmfit, 
                    newdata = nid,
                   times = seq(tmin, tmax, length.out = 11),
                    return_newdata = TRUE)
  Spred <- predict(jmfit, 
                 newdata = nid,
                 times = seq(tmin, tmax, length.out = 11),
                 process = "event",
                 return_newdata = TRUE)
  
  plot(Lpred, Spred, 
       ylab_long = paste0("ID_", id_num, "  log(amh)"))
  
  return(list(Lpred = Lpred,
              Spred = Spred,
              nid0 = nid0,
              nid = nid))
}
```
