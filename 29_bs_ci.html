pull_ci <- function(data, type, group) {
  results <- data %>%
  as.data.frame() %>%
  rownames_to_column(var = "Tstart") %>%
  mutate(type = type,
         group = group) %>%
  pivot_longer(cols = -c(Tstart, type, group),
               names_to = "Dt",
               values_to = "bs")
}
